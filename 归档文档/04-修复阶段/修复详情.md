## 技术实现细节

### 配置菜单切换实现
```javascript
// 配置菜单点击处理
document.addEventListener('click', function(e) {
  const configMenuItem = e.target.closest('.config-menu-item');
  if (configMenuItem) {
    e.preventDefault();
    const configType = configMenuItem.dataset.config;
    
    // 更新菜单激活状态
    document.querySelectorAll('.config-menu-item').forEach(item => {
      item.classList.remove('active');
      item.style.background = 'transparent';
      item.style.color = '#94A3B8';
    });
    configMenuItem.classList.add('active');
    configMenuItem.style.background = 'rgba(59, 130, 246, 0.2)';
    configMenuItem.style.color = '#E2E8F0';
    
    // 切换配置内容
    document.querySelectorAll('.config-section').forEach(section => {
      section.style.display = 'none';
    });
    const targetSection = document.querySelector(`.config-section[data-section="${configType}"]`);
    if (targetSection) {
      targetSection.style.display = 'block';
    }
    return;
  }
});
```
