function createNewExperiment() {
  const name = document.getElementById('newExpName').value.trim();
  const modelType = document.getElementById('newExpModel').value;
  
  if (!name) {
    showNotification('❌ 请填写实验名称', 'error');
    return;
  }
  
  const expData = {
    experimentId: `EXP-${String(VirtualDB.getData().nextId.experiments).padStart(4, '0')}`,
    name,
    modelType,
    accuracy: 0,
    loss: 0,
    f1Score: 0,
    status: '进行中',
    createTime: new Date().toISOString().slice(0, 16).replace('T', ' ')
  };
  
  VirtualDB.add('experiments', expData);
  VirtualDB.addAuditLog('创建实验', name, `创建实验: ${name}`, '低风险');
  showNotification(`✅ 实验 "${name}" 创建成功！`, 'success');
  closeModal();
  
  // 刷新当前页面
  Router.navigate('experiment');
}
```

function changePassword() {
  openModal('修改密码', `

        <input id="newPassword" type="password" style="width: 100%; padding: 8px; background: rgba(51, 65, 85, 0.5); border: 1px solid #334155; border-radius: 6px; color: #E2E8F0;">
      </div>
      <div>
        <label style="display: block; margin-bottom: 4px; font-size: 13px;">确认新密码</label>
        <input id="confirmPassword" type="password" style="width: 100%; padding: 8px; background: rgba(51, 65, 85, 0.5); border: 1px solid #334155; border-radius: 6px; color: #E2E8F0;">
      </div>
      <button class="btn btn-primary" onclick="submitPasswordChange()">确认修改</button>
    </div>
  `);
}

function submitPasswordChange() {
  const oldPwd = document.getElementById('oldPassword').value;
  const newPwd = document.getElementById('newPassword').value;
  const confirmPwd = document.getElementById('confirmPassword').value;
  
  if (!oldPwd || !newPwd || !confirmPwd) {
    showNotification('❌ 请填写所有字段', 'error');
    return;
  }
  
  if (newPwd !== confirmPwd) {
    showNotification('❌ 两次输入的新密码不一致', 'error');
    return;
  }
  
  if (newPwd.length < 6) {
    showNotification('❌ 密码长度不能少于6位', 'error');
    return;
  }
  
  // 模拟密码修改
  VirtualDB.addAuditLog('修改密码', 'admin', '用户修改了登录密码', '中风险');
  showNotification('✅ 密码修改成功！', 'success');
  closeModal();
}
```
