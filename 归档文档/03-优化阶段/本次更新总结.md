### 2. 拓扑排序执行
使用拓扑排序算法按依赖关系顺序执行节点，支持并行执行。

```javascript
// 找到起始节点
const startNodes = nodes.filter(n => !connections.some(c => c.to === n.id));

// 按依赖关系执行
while (queue.length > 0) {
  const node = queue.shift();
  // 执行节点
  await executeNode(node);
  // 添加下一个节点到队列
  addNextNodes(node);
}
```

### 3. 循环依赖检测
使用DFS算法检测流程中的环路。

```javascript
function hasCircularDependency() {
  const visited = new Set();
  const recursionStack = new Set();
  
  function dfs(nodeId) {
    visited.add(nodeId);
    recursionStack.add(nodeId);
    
    // 检查所有输出连接
    for (const conn of connections.filter(c => c.from === nodeId)) {
      if (!visited.has(conn.to)) {
        if (dfs(conn.to)) return true;
      } else if (recursionStack.has(conn.to)) {
        return true; // 发现环路
      }
    }
    
    recursionStack.delete(nodeId);
    return false;
  }
}
```

### 4. 实时搜索过滤
使用input事件实现实时搜索，无需点击按钮。

```javascript
searchInput.addEventListener('input', (e) => {
  const searchTerm = e.target.value.toLowerCase();
  rows.forEach(row => {
    const text = row.textContent.toLowerCase();
    row.style.display = text.includes(searchTerm) ? '' : 'none';
  });
});
```
